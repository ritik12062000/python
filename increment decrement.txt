#include <stdio.h>

int main() {

	printf("45 + 7 = %d\n", 45 + 7);
	printf("234 - 55 = %d\n", 234 - 55);
	printf("56 * 4 = %d\n", 56 * 4);
	printf("22 / 2 = %d\n", 22 / 2);

	printf("34 + 67 + 23 + 12 + 6 = %d\n", 34 + 67 + 23 + 12 + 6);
	printf("23 + 89 + 12 + 56 = %d\n", 23 + 89 + 12 + 56);

	printf("56 * 34 + 12 / 2 - 13 = %d\n", 56 * 34 + 12 / 2 - 13);

	printf("1.2 * 6.7 - 4.5 / 2 = %.3f\n", 1.2 * 6.7 - 4.5 / 2);
	printf("1.6 * (12 - 2) / 2 + 7.8 - 1.5 = %.3f\n",
	       1.6 * (12 - 2) / 2 + 7.8 - 1.5);

	printf("4.5 * 1.2 - 82 / 2 - 5.6 + 1.234 = %.3f\n",
	       4.5 * 1.2 - 82 / 2 - 5.6 + 1.234);

	printf("8 * 1 + (4 - 2) / 1 + 6 = %d\n", 8 * 1 + (4 - 2) / 1 + 6);

	printf("3.4 + 1.2 * (3.7 - 1.2) / 2 + 6.8 = %.3f\n",
	       3.4 + 1.2 * (3.7 - 1.2) / 2 + 6.8);

	printf("8.9 * 1.67 - (6.21 - 5.6) + 7.8 - 34 + 157 = %.3f\n",
	       8.9 * 1.67 - (6.21 - 5.6) + 7.8 - 34 + 157);

	return 0;
}
